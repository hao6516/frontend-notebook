# 数组去重

# 一、数组内容全部为基本类型

# 1.Set(Array.from和解构运算符都可)

function uniqueArray(arr) {
  // return Array.from(new Set(arr))
  return [...new Set(arr)] 
}

Set中使用 === 来判断相等

# 2.双重for循环

function uniqueArray(arr) {
  let len = arr.length
  for (let i = 0; i < len; i++) {
    for (let j = i + 1; j < len; j++) {
      if (arr[i] === arr[j]) {
        arr.splice(j, 1)
        len--
        j--
      }
    }
  }
}

# 3.indexOf

function uniqueArray(arr) {
  const newArr = []
  arr.forEach((item) => {
    newArr.indexOf(item) === -1 && newArr.push(item)
  })
  return newArr
}

# 4.includes

function uniqueArray(arr) {
  const newArr = []
  arr.forEach((item) => {
    !newArr.includes(item) && newArr.push(item)
  })
  return newArr
}

# 5.filter

function uniqueArray(arr) {
  return arr.filter((item, index) => arr.indexOf(item) === index)
}

# 6.reduce

function uniqueArray(arr) {
  return arr.reduce((pre, cur) => (pre.includes(cur) ? pre : [...pre, cur]), [])
}

# 7.Map

function uniqueArray(arr) {
  const newArr = []
  const map = new Map()
  arr.forEach((item) => {
    if (!map.has(item)) {
      map.set(item, true)
      newArr.push(item)
    }
  })
  return newArr
}

# 8.Object

function uniqueArray(arr) {
  const newArr = []
  const obj = {}
  arr.forEach((item) => {
    if (!obj[item]) {
      obj[item] = true
      newArr.push(item)
    }
  })
  return newArr
}

# 二、数组内容含对象

🌰[{a:1,b:2}, {b:2,a:1}] 此时，光比较地址这种浅比较是不行的，需要对对象作深层次的比较
比较算法:

const isObject = (val) => val !== null && typeof val === 'object'
function equals(val1, val2) {
  if (!isObject(val1) || !isObject(val2)) {
    return Object.is(val1, val2)
  }
  if (val1 === val2) return true
  const val1Keys = Object.keys(val1)
  const val2Keys = Object.keys(val2)
  if (val1Keys.length !== val2Keys.length) {
    return false
  }
  for (let key in val1) {
    if (!val2Keys.includes(key)) {
      return false
    }
    if (!equals(val1[key], val2[key])) {
      return false
    }
  }
  return true
}
